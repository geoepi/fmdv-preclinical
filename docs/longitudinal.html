Shedding Rates

Run model for shedding (nasal swabs)

Hide code 
return_quants <- c(0.025, 0.05, 0.25, 0.5, 0.75, 0.95, 0.975)

pc_prec_rw <- list(theta = list(prior="pc.prec", 
                                 param=c(1, 0.001)))

pc_prec_iid <- list(theta = list(prior="pc.prec", 
                                 param=c(3, 0.001)))

shed_form <- y_shed ~ 1 + f(time_step,
                             model = "rw1",
                             constr = TRUE,
                             scale.model = TRUE,
                             hyper=pc_prec_rw) +
                           f(animal,
                             model = "iid",
                             constr=TRUE,
                             hyper=pc_prec_iid) + 
                           f(exp_type, 
                             model = "iid",
                             constr=TRUE,
                             hyper=pc_prec_iid) +
                             temp 

shed_model <- inla(shed_form, 
                   data = longitud_df, 
                   family = "gaussian",
                   quantiles = return_quants,
                   verbose=FALSE,
                   control.compute=list(dic = TRUE, cpo = FALSE, waic = TRUE))

                                 param=c(1, 0.001)))

temper_form <- y_temper ~ -1 + f(time_step,
                                 model = "rw1",
                                 constr = TRUE,
                                 scale.model = TRUE,
                                 hyper=pc_prec_rw) +
                                f(animal,
                                 model = "iid",

                                 hyper=pc_prec_iid) + 
                                f(exp_type, 
                                 model = "iid",
                                 constr=TRUE,
                                 hyper=pc_prec_iid) +
                              score

temper_model <- inla(temper_form, 
                   data = longitud_df, 
                   family = "gaussian",
